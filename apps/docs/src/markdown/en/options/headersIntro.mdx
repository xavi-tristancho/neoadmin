These are the general options of the header.

### Example

```js
const headers = {
  type: "CRUD",
    // highlight-start
  options: {
    name: "Product",
    primaryKey: "reference",
    requests: {...},
    route: {...},
    tableOptions: {...},
    upsertOptions: {...},
  },
    // highlight-end
  sections: [...],
};

export default headers;
```
