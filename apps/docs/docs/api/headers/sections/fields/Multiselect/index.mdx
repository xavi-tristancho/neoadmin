The Multiselect is a type of `field`. Is the representation of a `select` type of input that can have multiple options selected if needed.

### Example

```js
import { CustomFieldsContainer } from "components";

const header = {
  type: "CRUD",
  options: {...},
  sections: [
    {
      fields: [
        // only 1 possible value selected
        // highlight-start
        {
          label: "Category",
          property: "category",
          type: "multiselect",
          relation: {
            isMulti: false,
            options: [
              { id: 0, name: "Desserts" },
              { id: 1, name: "Well-being" },
              { id: 2, name: "Travels" },
              { id: 3, name: "Gardening" },
              { id: 4, name: "Sports" },
              { id: 5, name: "Gadgets" },
            ],
            format: (category: Category) => category.name,
          },
          tableOptions: {
            format: (category: Category) => category.name,
          },
        },
        // highlight-end
        // more than 1 possible values selected
        // highlight-start
        {
          label: "Category",
          property: "category",
          type: "multiselect",
          relation: {
            isMulti: true,
            options: [
              { id: 0, name: "Desserts" },
              { id: 1, name: "Well-being" },
              { id: 2, name: "Travels" },
              { id: 3, name: "Gardening" },
              { id: 4, name: "Sports" },
              { id: 5, name: "Gadgets" },
            ],
            format: (category: Category) => category.name,
          },
          tableOptions: {
            format: (category: Category) => category.name,
          },
        },
      // highlight-end
      ],
    }
  ],
};

export default headers;
```
