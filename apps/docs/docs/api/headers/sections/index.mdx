# Sections

The `sections` property is an array of objects, each object is meant to be a `section`.

```js
import { CustomFieldsContainer } from "components";

const headers = {
  type: "CRUD",
  options: {...},
    // highlight-start
  sections: [
    {
      // first section
      component: () => {
        return <div>This is a section custom component</div>;
      },
      fields: [...],
      FieldsContainer: ({ children, style }) => (
        <CustomFieldsContainer style={{ ...style, background: "white", padding: "2rem" }}>
          {children}
        </CustomFieldsContainer>
      ),
      fieldsContainerStyles: { background: "white", padding: "2rem" },
      //Subtitle: "function<component>" // ??,
      subtitle: "Subtitle first section",
      //Title: "function<component>" // ??,
      title: "Title first section",
    },
    {
      // second section
    }
  ],
    // highlight-end
};

export default headers;
```
